## Part 2: Reusable Workflow

### Step 2.1: Create Reusable Docker Build
```yaml
# .github/workflows/reusable-docker-build.yml
name: Reusable Docker Build

on:
  workflow_call:
    inputs:
      service-name:
        required: true
        type: string
      service-path:
        required: true
        type: string
      aws-region:
        required: false
        type: string
        default: 'ap-south-1'
    secrets:
      AWS_ROLE_ARN:
        required: true

env:
  AWS_REGION: ${{ inputs.aws-region }}

jobs:
  build:
    runs-on: ubuntu-latest
    permissions:
      id-token: write
      contents: read
      
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Configure AWS credentials (OIDC)
        uses: aws-actions/configure-aws-credentials@v4
        with:
          role-to-assume: ${{ secrets.AWS_ROLE_ARN }}
          aws-region: ${{ env.AWS_REGION }}

      - name: Login to ECR
        id: login-ecr
        uses: aws-actions/amazon-ecr-login@v2

      - name: Set up Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Extract metadata
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: ${{ steps.login-ecr.outputs.registry }}/techitfactory/${{ inputs.service-name }}
          tags: |
            type=sha,prefix=
            type=ref,event=branch
            type=ref,event=pr
            type=raw,value=latest,enable=${{ github.ref == 'refs/heads/main' }}

      - name: Build and push
        uses: docker/build-push-action@v5
        with:
          context: ${{ inputs.service-path }}
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          cache-from: type=gha
          cache-to: type=gha,mode=max

      - name: Run Trivy scan
        uses: aquasecurity/trivy-action@master
        with:
          image-ref: '${{ steps.login-ecr.outputs.registry }}/techitfactory/${{ inputs.service-name }}:${{ github.sha }}'
          format: 'sarif'
          output: 'trivy-results.sarif'
          severity: 'CRITICAL,HIGH'
          exit-code: '1'
        continue-on-error: true

      - name: Upload Trivy results
        uses: github/codeql-action/upload-sarif@v2
        if: always()
        with:
          sarif_file: 'trivy-results.sarif'
```

## Part 3: Per-Service Workflows

### Step 3.1: API Gateway Workflow
```yaml
# .github/workflows/ci-api-gateway.yml
name: CI - API Gateway

on:
  push:
    branches: [main]
    paths:
      - 'services/api-gateway/**'
      - '.github/workflows/ci-api-gateway.yml'
  pull_request:
    branches: [main]
    paths:
      - 'services/api-gateway/**'

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'
          cache-dependency-path: services/api-gateway/package-lock.json
      - run: cd services/api-gateway && npm ci
      - run: cd services/api-gateway && npm test --if-present
      - run: cd services/api-gateway && npm run lint --if-present

  build:
    needs: test
    if: github.event_name == 'push' && github.ref == 'refs/heads/main'
    uses: ./.github/workflows/reusable-docker-build.yml
    with:
      service-name: api-gateway
      service-path: services/api-gateway
    secrets:
      AWS_ROLE_ARN: ${{ secrets.AWS_ROLE_ARN }}

  update-gitops:
    needs: build
    runs-on: ubuntu-latest
    steps:
      - name: Checkout GitOps repo
        uses: actions/checkout@v4
        with:
          repository: TechITFactory/techitfactory-gitops
          token: ${{ secrets.GITOPS_TOKEN }}
          path: gitops

      - name: Update image tag
        run: |
          cd gitops/environments/dev/api-gateway
          sed -i "s|newTag:.*|newTag: ${{ github.sha }}|" kustomization.yaml

      - name: Commit and push
        run: |
          cd gitops
          git config user.name "GitHub Actions"
          git config user.email "actions@github.com"
          git add .
          git commit -m "Update api-gateway to ${{ github.sha }}" || exit 0
          git push
```

## Part 4: Secrets Configuration

### Required GitHub Secrets
```
AWS_ROLE_ARN       = arn:aws:iam::ACCOUNT:role/techitfactory-github-terraform
GITOPS_TOKEN       = <Personal Access Token with repo scope>
```

### How to Create GITOPS_TOKEN
1. Go to GitHub → Settings → Developer settings → Personal access tokens
2. Generate new token (classic)
3. Select scopes: `repo` (full control)
4. Copy token and add as secret in techitfactory-app repo
